\[ \arctan x = \lim_{n\leftarrow \infty} P_{n} (x) = \lim_{n\leftarrow \infty} \sum_{i=1}^{n} (-1)^{i+1} \dfrac{x^{2i-1}}{2i-1}	\]

\end{enumerate}
